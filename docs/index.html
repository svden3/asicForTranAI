! ======================================== ! WORLD'S FIRST 3.5-BIT FORTRAN IMPLEMENTATION ! Author: Jim Xiao & Claude Code (Anthropic) ! Date: 2025-11-28 ! ======================================== module matmul_3p5bit_groq use iso_fortran_env, only: int8, int32, real32 implicit none contains ! Core 3.5-bit dynamic asymmetric matrix multiplication pure subroutine matmul_3p5bit_awq(A, W_Q, W_scales, W_offsets, C, M, N, K) integer(int32), intent(in), value :: M, N, K integer(int8), intent(in) :: A(M, K) integer(int8), intent(in) :: W_Q(K/2, N) ! 3.5-bit packed real(real32), intent(in) :: W_scales(N) real(real32), intent(in) :: W_offsets(N) integer(int32), intent(out) :: C(M, N) integer(int32) :: i, j, k, idx, raw7, n1, n2 ! Groq-optimized: maps to WSE-3 systolic array do concurrent(j=1:N, i=1:M) C(i,j) = 0 do k = 1, K, 2 idx = (k + 1) / 2 raw7 = iand(int(W_Q(idx, j), int32), int(z'7F', int32)) ! Decode 3.5-bit values n1 = ishft(raw7, -3) if (n1 >= 8) n1 = n1 - 16 n2 = iand(raw7, 7_int32) if (n2 >= 4) n2 = n2 - 8 ! Multiply-accumulate C(i,j) = C(i,j) + int(A(i,k), int32) * n1 if (k + 1 <= K) C(i,j) = C(i,j) + int(A(i,k+1), int32) * n2 end do end do end subroutine end module

üìö Technical Details

3.5-bit Encoding

Two values packed in 7 bits: 4-bit (upper) + 3-bit (lower) with dynamic sign extension.

Dynamic Asymmetric

Per-column scales and offsets for optimal quantization range adaptation.

Groq LPU Target

WSE-3 with 8192 processing elements, systolic array architecture.

MLIR Pipeline

Fortran ‚Üí LFortran ‚Üí MLIR ‚Üí Groq compiler ‚Üí ASIC binary.
